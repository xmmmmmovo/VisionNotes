J_history = [] # 
def J(theta, X, y): # J
    return (np.sum(np.square(X.dot(theta) - y)) / (2 * len(y)))

# DJ
def dJ(theta, X, y):
#     print(theta.shape)
#     print(X)
    return X.T.dot(X.dot(theta) - y) / len(y)

def gradient_descent(X, y, alpha, steps):
    theta = np.zeros(X.shape[1])
    for s in range(steps):
        gd = dJ(theta, X, y)
        temp_theta = theta
        theta = theta - gd * alpha
        J_history.append(theta)   
    return theta
